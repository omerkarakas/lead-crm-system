```typescript
import { WhatsAppMessage } from '@/types';
import pb from '@/lib/pocketbase';

// Fetch all messages for a lead
export async function getLeadWhatsAppMessages(leadId: string): Promise<WhatsAppMessage[]> {
  const records = await pb.collection('whatsapp_messages')
    .getList(1, 100, {
      filter: `lead_id = "${leadId}"`,
      sort: 'created_at'
    });

  return records.items;
}

// Get message count for badge
export async function getUnreadMessageCount(leadId: string): Promise<number> {
  const records = await pb.collection('whatsapp_messages')
    .getList(1, 1, {
      filter: `lead_id = "${leadId}" && direction = "incoming" && status != "read"`
    });

  return records.totalItems;
}
```

### Task 2: Create chat bubble components

**File:** `components/leads/ChatBubble.tsx`
```typescript
import { WhatsAppMessage } from '@/types';

interface ChatBubbleProps {
  message: WhatsAppMessage;
}

export function ChatBubble({ message }: ChatBubbleProps) {
  const isIncoming = message.direction === 'incoming';

  return (
    <div className={`flex ${isIncoming ? 'justify-start' : 'justify-end'}`}>
      <div
        className={`max-w-[80%] rounded-2xl px-4 py-2 ${
          isIncoming
            ? 'bg-gray-100 text-gray-900 rounded-tl-none'
            : 'bg-green-500 text-white rounded-tr-none'
        }`}
      >
        <p className="text-sm whitespace-pre-wrap">{message.message_text}</p>

        <div className={`flex items-center gap-2 mt-1 text-xs ${
          isIncoming ? 'text-gray-500' : 'text-green-100'
        }`}>
          {/* Timestamp */}
          {message.sent_at && (
            <span>
              {new Date(message.sent_at).toLocaleTimeString('tr-TR', {
                hour: '2-digit',
                minute: '2-digit'
              })}
            </span>
          )}

          {/* Status badge (only for outgoing) */}
          {!isIncoming && (
            <MessageStatusBadge status={message.status} />
          )}
        </div>
      </div>
    </div>
  );
}

function MessageStatusBadge({ status }: { status: WhatsAppMessage['status'] }) {
  const variants = {
    sent: 'text-yellow-200',
    delivered: 'text-blue-200',
    read: 'text-green-200',
    failed: 'text-red-200'
  };

  const icons = {
    sent: '⏳',
    delivered: '✓',
    read: '✓✓',
    failed: '✗'
  };

  return (
    <span className={variants[status]} title={status}>
      {icons[status]}
    </span>
  );
}
```

**File:** `components/leads/WhatsAppConversation.tsx`
```typescript
'use client';

import { useEffect, useState } from 'react';
import { WhatsAppMessage } from '@/types';
import { getLeadWhatsAppMessages } from '@/lib/api/whatsapp';
import { ChatBubble } from './ChatBubble';
import { MessageSquare } from 'lucide-react';

interface WhatsAppConversationProps {
  leadId: string;
}

export function WhatsAppConversation({ leadId }: WhatsAppConversationProps) {
  const [messages, setMessages] = useState<WhatsAppMessage[]>([]);
  const [loading, setLoading] = useState(true);

  const fetchMessages = async () => {
    setLoading(true);
    try {
      const data = await getLeadWhatsAppMessages(leadId);
      setMessages(data);
    } catch (error) {
      console.error('Failed to fetch messages:', error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchMessages();

    // Auto-refresh every 30 seconds
    const interval = setInterval(fetchMessages, 30000);
    return () => clearInterval(interval);
  }, [leadId]);

  if (loading) {
    return (
      <div className="flex items-center justify-center py-8 text-gray-500">
        Yükleniyor...
      </div>
    );
  }

  if (messages.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-8 text-gray-500">
        <MessageSquare className="w-12 h-12 mb-2 opacity-50" />
        <p>Henüz mesaj yok.</p>
      </div>
    );
  }

  return (
    <div className="space-y-3">
      {messages.map((message) => (
        <ChatBubble key={message.id} message={message} />
      ))}
    </div>
  );
}
```
