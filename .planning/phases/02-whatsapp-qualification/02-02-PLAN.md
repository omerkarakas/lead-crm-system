### Task 2: Create background job for delayed poll sending

**File:** `lib/api/leads.ts` — Add function:
```typescript
// Send poll after 1 minute delay
export async function sendPollAfterDelay(leadId: string): Promise<void> {
  // 1 minute delay
  await new Promise(resolve => setTimeout(resolve, 60000));

  // Fetch lead
  const lead = await fetchLead(leadId);
  if (!lead || !lead.phone) {
    console.error('Lead not found or no phone:', leadId);
    return;
  }

  // Fetch active questions
  const questions = await fetchActiveQuestions();
  if (questions.length === 0) {
    console.error('No active questions found');
    return;
  }

  // Format poll message
  const pollMessage = formatPollMessage(lead, questions);

  // Send WhatsApp
  const chatId = lead.phone.replace(/\D/g, '') + '@c.us';
  const result = await sendWhatsAppMessage(chatId, pollMessage);

  // Log message
  await logWhatsAppMessage({
    lead_id: leadId,
    direction: 'outgoing',
    message_text: pollMessage,
    message_type: 'poll',
    status: 'sent',
    sent_at: new Date().toISOString(),
    green_api_id: result.idMessage
  });

  // Update lead: qa_sent = true
  await updateLead(leadId, { qa_sent: true, qa_sent_at: new Date().toISOString() });
}
```

### Task 3: Create poll message formatter

**File:** `lib/whatsapp/message-formatter.ts`

```typescript
import { QAQuestion, Lead } from '@/types';
import { WELCOME_MESSAGE, POLL_FOOTER } from '@/config/qa';

export function formatPollMessage(lead: Lead, questions: QAQuestion[]): string {
  // Replace variables in welcome message
  let welcome = WELCOME_MESSAGE.replace('{name}', lead.name || 'Değerli Müşterimiz');
  if (lead.company) {
    welcome = welcome.replace('{company}', lead.company);
  }

  // Format questions
  const questionsText = questions.map((q, index) => {
    const num = index + 1;
    const options = q.options.join('\n   ');
    return `${num}. ${q.question_text}\n   ${options}`;
  }).join('\n\n');

  return welcome + '\n\n' + questionsText + POLL_FOOTER;
}

export function formatBookingLinkMessage(meetingUrl: string): string {
  return `🎉 Harika! Randevunuzu oluşturmak için aşağıdaki linki kullanabilirsiniz:\n\n${meetingUrl}\n\nGörüşmek üzere!`;
}

export function formatLowQualityMessage(): string {
  return `Teşekkürler! Başvurunuz incelendi. En kısa sürede size dönüş yapacağız. 🙏`;
}
```

### Task 4: Create WhatsApp webhook endpoint

**File:** `app/api/whatsapp/webhook/route.ts`
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { findLeadByPhone } from '@/lib/api/leads';
import { parsePollAnswer } from '@/lib/whatsapp/poll-parser';
import { saveAnswer, calculateTotalScore } from '@/lib/api/qa';
import { logWhatsAppMessage } from '@/lib/api/whatsapp';
import {
  formatBookingLinkMessage,
  formatLowQualityMessage
} from '@/lib/whatsapp/message-formatter';

export async function POST(req: NextRequest) {
  try {
    const body = await req.json();

    // Extract phone and message from Green API webhook
    const senderData = body.body?.senderData || body.senderData || {};
    const messageData = body.body?.messageData || body.messageData || {};

    let chatId = senderData.chatId || senderData.sender || '';
    let phone = chatId.replace('@c.us', '');

    let messageText = '';
    if (messageData.textMessageData) {
      messageText = messageData.textMessageData.textMessage || '';
    } else if (messageData.extendedTextMessageData) {
      messageText = messageData.extendedTextMessageData.text || '';
    }

    // Log incoming message
    await logWhatsAppMessage({
      lead_id: '', // Will fill after finding lead
      direction: 'incoming',
      message_text: messageText,
      message_type: 'poll',
      status: 'received',
      sent_at: new Date().toISOString()
    });

    // Find lead by phone
    const lead = await findLeadByPhone(phone);
    if (!lead) {
      // Unknown sender - could log as unknown
      return NextResponse.json({ success: false, message: 'Unknown sender' });
    }

    // Parse answer
    const answer = parsePollAnswer(messageText);
    if (!answer) {
      // Invalid format - send retry message
      await sendWhatsAppMessage(
        phone.replace(/\D/g, '') + '@c.us',
        'Cevap formatı hatalı. Lütfen "1a, 2b" formatında yazın.'
      );
      return NextResponse.json({ success: false, message: 'Invalid format' });
    }

    // Save answers and calculate score
    let totalScore = lead.total_score || 0;
    for (const [questionIndex, selectedOption] of Object.entries(answer)) {
      const qIndex = parseInt(questionIndex);
      const questions = await fetchActiveQuestions();

      if (qIndex < questions.length) {
        const question = questions[qIndex];
        const points = question.points[selectedOption] || 0;

        await saveAnswer({
          lead_id: lead.id,
          question_id: question.id,
          selected_answer: selectedOption,
          points_earned: points,
          answered_at: new Date().toISOString()
        });

        totalScore += points;
      }
    }

    // Update lead score
    const quality = totalScore >= QUALITY_SCORE_THRESHOLD ? 'qualified' : 'pending';
    await updateLead(lead.id, {
      total_score: totalScore,
      quality: quality,
      qa_completed: true,
      qa_completed_at: new Date().toISOString()
    });

    // Send response based on score
    const chatId = phone.replace(/\D/g, '') + '@c.us';
    let responseMessage = '';

    if (quality === 'qualified') {
      responseMessage = formatBookingLinkMessage(CALCOM_MEETING_URL);
    } else {
      responseMessage = formatLowQualityMessage();
    }

    await sendWhatsAppMessage(chatId, responseMessage);
    await logWhatsAppMessage({
      lead_id: lead.id,
      direction: 'outgoing',
      message_text: responseMessage,
      message_type: quality === 'qualified' ? 'booking_link' : 'info',
      status: 'sent',
      sent_at: new Date().toISOString()
    });

    return NextResponse.json({ success: true, score: totalScore, quality });

  } catch (error) {
    console.error('WhatsApp webhook error:', error);
    return NextResponse.json(
      { success: false, message: 'Internal error' },
      { status: 500 }
    );
  }
}
```

### Task 5: Create poll answer parser

```typescript
// Parse answers like "1a, 2b", "a b", "ab", "1a2b"
export function parsePollAnswer(message: string): Record<number, string> | null {
  const clean = message.toLowerCase().trim().replace(/[^a-z0-9]/gi, '');

  // Format 1: "1a2b" -> {0: "a", 1: "b"}
  // Format 2: "ab" -> {0: "a", 1: "b"} (implicit order)

  const result: Record<number, string> = {};

  // Try format 1: "1a2b"
  const format1Regex = /(\d)([abc])/g;
  let match;
  let found = false;
  while ((match = format1Regex.exec(clean)) !== null) {
    const questionIndex = parseInt(match[1]) - 1; // 1-indexed to 0-indexed
    const answer = match[2];
    result[questionIndex] = answer;
    found = true;
  }

  if (found) return result;

  // Try format 2: "ab" (implicit order)
  if (/^[abc]+$/.test(clean)) {
    for (let i = 0; i < clean.length; i++) {
      result[i] = clean[i];
    }
    return result;
  }

  return null; // Invalid format
}
```

```typescript
export async function sendWhatsAppMessage(chatId: string, message: string) {
  const GREEN_API_INSTANCE_ID = process.env.GREEN_API_INSTANCE_ID;
  const GREEN_API_TOKEN = process.env.GREEN_API_TOKEN;

  const response = await fetch(
    `https://api.green-api.com/waInstance${GREEN_API_INSTANCE_ID}/sendMessage/${GREEN_API_TOKEN}`,
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ chatId, message })
    }
  );

  return await response.json();
}
```
