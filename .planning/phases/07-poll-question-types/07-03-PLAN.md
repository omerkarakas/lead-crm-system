export function parsePollAnswer(
  message: string,
  questionType: 'single' | 'multiple' | 'likert' | 'open'
): Record<number, string | string[]> | string | null {
  if (!message || typeof message !== 'string') {
    return null;
  }

  const clean = message.toLowerCase().trim().replace(/[^a-z0-9]/gi, '');

  if (clean.length === 0) {
    return null;
  }

  // Open-ended: return as-is (will be validated by length)
  if (questionType === 'open') {
    return message.trim();
  }

  // Single choice: "1a, 2b" or "ab"
  if (questionType === 'single') {
    return parseSingleChoiceAnswer(clean);
  }

  // Multiple choice: "1ab" (Q1: a and b) or "1abc" (Q1: a, b, c)
  if (questionType === 'multiple') {
    return parseMultipleChoiceAnswer(clean);
  }

  // Likert: "13" (Q1: 1, Q2: 3)
  if (questionType === 'likert') {
    return parseLikertAnswer(clean);
  }

  return null;
}

function parseSingleChoiceAnswer(clean: string): Record<number, string> | null {
  // Existing logic (keep for backward compatibility)
  const result: Record<number, string> = {};

  // Try format 1: "1a2b" (numbered with answers)
  const format1Regex = /(\d)([a-z])/g;
  let match;
  let found = false;
  while ((match = format1Regex.exec(clean)) !== null) {
    const questionIndex = parseInt(match[1]) - 1;
    const answer = match[2];
    result[questionIndex] = answer;
    found = true;
  }

  if (found && Object.keys(result).length > 0) {
    return result;
  }

  // Try format 2: "ab" (implicit order - a for Q1, b for Q2)
  if (/^[a-z]+$/.test(clean)) {
    for (let i = 0; i < clean.length; i++) {
      result[i] = clean[i];
    }
    return result;
  }

  return null;
}

function parseMultipleChoiceAnswer(clean: string): Record<number, string[]> | null {
  const result: Record<number, string[]> = {};

  // Format: "1ab" means question 1, answers a and b
  // Format: "1ab2c" means question 1: a,b and question 2: c
  const formatRegex = /(\d+)([a-z]+)/g;
  let match;
  let found = false;

  while ((match = formatRegex.exec(clean)) !== null) {
    const questionIndex = parseInt(match[1]) - 1;
    const answers = match[2].split(''); // "ab" -> ["a", "b"]
    result[questionIndex] = answers;
    found = true;
  }

  if (found && Object.keys(result).length > 0) {
    return result;
  }

  return null;
}

function parseLikertAnswer(clean: string): Record<number, string> | null {
  const result: Record<number, string> = {};

  // Format: "13" means question 1: 1, question 2: 3
  // Format: "1-3-5" means question 1: 1, question 2: 3, question 3: 5
  const format1Regex = /(\d)[\-]?(\d)/g;
  let match;
  let found = false;

  while ((match = format1Regex.exec(clean)) !== null) {
    const questionIndex = parseInt(match[1]) - 1;
    const rating = match[2];
    if (parseInt(rating) >= 1 && parseInt(rating) <= 5) {
      result[questionIndex] = rating;
      found = true;
    }
  }

  if (found && Object.keys(result).length > 0) {
    return result;
  }

  // Fallback: sequential digits "13" -> Q1: 1, Q2: 3
  if (/^\d+$/.test(clean)) {
    let qIndex = 0;
    for (let i = 0; i < clean.length; i++) {
      const digit = clean[i];
      if (parseInt(digit) >= 1 && parseInt(digit) <= 5) {
        result[qIndex++] = digit;
      }
    }
    if (Object.keys(result).length > 0) {
      return result;
    }
  }

  return null;
}

export function validateAnswers(
  answers: Record<number, string | string[]> | string | null,
  questionType: string,
  questionCount: number,
  question?: QAQuestion
): boolean {
  if (!answers) {
    return false;
  }

  // Open-ended: validate length
  if (questionType === 'open') {
    if (typeof answers !== 'string') {
      return false;
    }
    const min = question?.min_length || 10;
    const max = question?.max_length || 1000;
    return answers.length >= min && answers.length <= max;
  }

  // Single/Multiple/Likert: validate question indices and answers
  if (typeof answers !== 'object') {
    return false;
  }

  for (const [questionIndex, answer] of Object.entries(answers)) {
    const qIndex = parseInt(questionIndex);

    if (qIndex < 0 || qIndex >= questionCount) {
      return false;
    }

    // Single choice: validate a-z
    if (questionType === 'single') {
      if (typeof answer !== 'string' || !/^[a-z]$/.test(answer)) {
        return false;
      }
    }

    // Multiple choice: validate array of a-z
    if (questionType === 'multiple') {
      if (!Array.isArray(answer) || answer.length === 0) {
        return false;
      }
      if (!answer.every(a => /^[a-z]$/.test(a))) {
        return false;
      }
    }

    // Likert: validate 1-5
    if (questionType === 'likert') {
      if (typeof answer !== 'string') {
        return false;
      }
      const rating = parseInt(answer);
      if (isNaN(rating) || rating < 1 || rating > 5) {
        return false;
      }
    }
  }

  return true;
}

export function formatAnswersForDisplay(
  answers: Record<number, string | string[]> | string | null
): string {
  if (!answers) {
    return '';
  }

  // Open-ended
  if (typeof answers === 'string') {
    return answers;
  }

  // Structured answers
  const sortedEntries = Object.entries(answers)
    .sort(([a], [b]) => parseInt(a) - parseInt(b));

  return sortedEntries
    .map(([qIndex, answer]) => {
      const qNum = parseInt(qIndex) + 1;
      if (Array.isArray(answer)) {
        return `${qNum}${answer.join('')}`; // "1ab"
      }
      return `${qNum}${answer}`; // "1a"
    })
    .join(', ');
}

export function calculateAnswerScore(
  question: QAQuestion,
  answer: string | string[]
): number {
  if (question.question_type === 'single') {
    // Single choice: lookup points mapping
    if (typeof answer !== 'string') {
      return 0;
    }
    return question.points?.[answer] || 0;
  }

  if (question.question_type === 'multiple') {
    // Multiple choice: sum of all selected option points
    if (!Array.isArray(answer)) {
      return 0;
    }
    return answer.reduce((total, selected) => {
      return total + (question.points?.[selected] || 0);
    }, 0);
  }

  if (question.question_type === 'likert') {
    // Likert: rating × points_per_scale
    const rating = parseInt(answer as string);
    if (isNaN(rating)) {
      return 0;
    }
    return rating * (question.points as number);
  }

  if (question.question_type === 'open') {
    // Open-ended: auto-score based on length
    if (typeof answer !== 'string') {
      return 0;
    }

    const { base_points, max_points, min_length, max_length } = question;
    const min = min_length || 10;
    const max = max_length || 1000;

    // Below minimum: 0 points
    if (answer.length < min) {
      return 0;
    }

    // Above maximum: cap at max
    if (answer.length >= max) {
      return max_points;
    }

    // Between min and max: scale base_points to max_points
    const range = max - min;
    const position = (answer.length - min) / range; // 0 to 1
    const bonus = Math.round(position * (max_points - base_points));
    return base_points + bonus;
  }

  return 0;
}

export async function processLeadAnswers(
  leadId: string,
  answers: Record<number, string | string[]> | string,
  questions: QAQuestion[]
): Promise<{ totalScore: number; answersSaved: QAAnswer[] }> {
  let totalScore = 0;
  const answersSaved: QAAnswer[] = [];

  // Open-ended: single question
  if (typeof answers === 'string') {
    const question = questions[0];
    if (question.question_type === 'open') {
      const points = calculateAnswerScore(question, answers);
      const saved = await saveAnswer({
        lead_id: leadId,
        question_id: question.id,
        answer_text: answers,
        points_earned: points,
        answered_at: new Date().toISOString()
      });
      answersSaved.push(saved);
      totalScore += points;
    }
    return { totalScore, answersSaved };
  }

  // Structured answers: map to questions
  for (const [qIndex, answerValue] of Object.entries(answers)) {
    const index = parseInt(qIndex);
    const question = questions[index];

    if (!question) {
      continue; // Skip invalid question index
    }

    const points = calculateAnswerScore(question, answerValue);
    const saved = await saveAnswer({
      lead_id: leadId,
      question_id: question.id,
      selected_answer: answerValue,
      points_earned: points,
      answered_at: new Date().toISOString()
    });
    answersSaved.push(saved);
    totalScore += points;
  }

  return { totalScore, answersSaved };
}

export function formatPollMessage(
  questions: Array<{ question_text: string; question_type: string; options?: string[]; order: number }>
): string {
  let message = "";

  for (const question of questions) {
    message += `*${question.order}. ${question.question_text}*\n\n`;

    if (question.question_type === 'single' || question.question_type === 'multiple') {
      // Show options as a) b) c) ...
      message += question.options?.join("\n") || "";
      if (question.question_type === 'multiple') {
        message += "\n(Birden fazla seçebilirsiniz - örn: 1ab)";
      } else {
        message += "\n(Cevap: 1a, 2b...)";
      }
    } else if (question.question_type === 'likert') {
      // Show Likert scale
      message += "1) Çok kötü\n2) Kötü\n3) Nötr\n4) İyi\n5) Çok iyi\n(Cevap: 1-5 arası sayı)";
    } else if (question.question_type === 'open') {
      // Open-ended instruction
      message += "Cevabınızı yazın...";
    }

    message += "\n\n";
  }

  return message;
}
