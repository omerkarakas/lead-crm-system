```typescript
export function validateQuestionData(data: CreateQAQuestionDto): { valid: boolean; error?: string } {
  // Validate single choice
  if (data.question_type === 'single') {
    if (!data.options || data.options.length < 2) {
      return { valid: false, error: 'Single choice questions require at least 2 options' };
    }
    if (!data.points || typeof data.points !== 'object') {
      return { valid: false, error: 'Single choice questions require points mapping' };
    }
  }

  // Validate multiple choice
  if (data.question_type === 'multiple') {
    if (!data.options || data.options.length < 2) {
      return { valid: false, error: 'Multiple choice questions require at least 2 options' };
    }
    if (!data.points || typeof data.points !== 'object') {
      return { valid: false, error: 'Multiple choice questions require points mapping' };
    }
    if (data.max_selections && data.max_selections > data.options.length) {
      return { valid: false, error: 'max_selections cannot exceed number of options' };
    }
  }

  // Validate Likert scale
  if (data.question_type === 'likert') {
    if (typeof data.points !== 'number' || data.points <= 0) {
      return { valid: false, error: 'Likert questions require numeric points value' };
    }
  }

  // Validate open-ended
  if (data.question_type === 'open') {
    if (!data.base_points || !data.max_points) {
      return { valid: false, error: 'Open questions require base_points and max_points' };
    }
  }

  return { valid: true };
}
```

2. Add answer validation based on question type:
```typescript
export function validateAnswerData(
  question: QAQuestion,
  answer: any
): { valid: boolean; error?: string } {
  if (question.question_type === 'single') {
    if (typeof answer.selected_answer !== 'string') {
      return { valid: false, error: 'Single choice requires string answer' };
    }
    if (!question.points[answer.selected_answer]) {
      return { valid: false, error: 'Invalid answer option' };
    }
  }

  if (question.question_type === 'multiple') {
    if (!Array.isArray(answer.selected_answer)) {
      return { valid: false, error: 'Multiple choice requires array of answers' };
    }
    if (answer.selected_answer.length === 0) {
      return { valid: false, error: 'At least one option must be selected' };
    }
  }

  if (question.question_type === 'likert') {
    const value = parseInt(answer.selected_answer);
    const min = question.scale_min || 1;
    const max = question.scale_max || 5;
    if (isNaN(value) || value < min || value > max) {
      return { valid: false, error: `Likert answer must be between ${min} and ${max}` };
    }
  }

  if (question.question_type === 'open') {
    if (typeof answer.answer_text !== 'string') {
      return { valid: false, error: 'Open question requires string answer' };
    }
    const min = question.min_length || 10;
    const max = question.max_length || 1000;
    if (answer.answer_text.length < min || answer.answer_text.length > max) {
      return { valid: false, error: `Answer must be between ${min} and ${max} characters` };
    }
  }

  return { valid: true };
}
```
