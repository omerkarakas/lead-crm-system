    1. Define permission types (src/types/permissions.ts):
       ```typescript
       export enum Role {
         ADMIN = 'admin',
         SALES = 'sales',
         MARKETING = 'marketing'
       }

       export enum Permission {
         // User management
         MANAGE_USERS = 'manage_users',

         // Lead permissions
         VIEW_ALL_LEADS = 'view_all_leads',
         VIEW_OWN_LEADS = 'view_own_leads',
         CREATE_LEADS = 'create_leads',
         EDIT_LEADS = 'edit_leads',
         DELETE_LEADS = 'delete_leads',

         // System permissions
         MANAGE_TAGS = 'manage_tags',
         VIEW_REPORTS = 'view_reports'
       }

       export const ROLE_PERMISSIONS: Record<Role, Permission[]> = {
         [Role.ADMIN]: [
           Permission.MANAGE_USERS,
           Permission.VIEW_ALL_LEADS,
           Permission.CREATE_LEADS,
           Permission.EDIT_LEADS,
           Permission.DELETE_LEADS,
           Permission.MANAGE_TAGS,
           Permission.VIEW_REPORTS
         ],
         [Role.SALES]: [
           Permission.VIEW_ALL_LEADS, // Sales can see all leads
           Permission.CREATE_LEADS,
           Permission.EDIT_LEADS,
           Permission.VIEW_REPORTS
         ],
         [Role.MARKETING]: [
           Permission.VIEW_ALL_LEADS, // Marketing can see all leads for campaigns
           Permission.CREATE_LEADS,
           Permission.MANAGE_TAGS,
           Permission.VIEW_REPORTS
         ]
       }
       ```

    2. Create permission composable (src/composables/usePermission.ts):
       ```typescript
       export function usePermission() {
         const authStore = useAuthStore()

         const hasRole = (role: Role): boolean => {
           return authStore.user?.role === role
         }

         const hasPermission = (permission: Permission): boolean => {
           if (!authStore.user) return false
           const userPermissions = ROLE_PERMISSIONS[authStore.user.role as Role]
           return userPermissions?.includes(permission) ?? false
         }

         const canManageUsers = computed(() => hasPermission(Permission.MANAGE_USERS))
         const canViewAllLeads = computed(() => hasPermission(Permission.VIEW_ALL_LEADS))
         const canCreateLeads = computed(() => hasPermission(Permission.CREATE_LEADS))
         const canEditLeads = computed(() => hasPermission(Permission.EDIT_LEADS))
         const canDeleteLeads = computed(() => hasPermission(Permission.DELETE_LEADS))
         const canManageTags = computed(() => hasPermission(Permission.MANAGE_TAGS))

         return {
           hasRole,
           hasPermission,
           canManageUsers,
           canViewAllLeads,
           canCreateLeads,
           canEditLeads,
           canDeleteLeads,
           canManageTags
         }
       }
       ```

    3. Add permission helpers to authStore:
       - Add isAdmin, isSales, isMarketing computed properties
       - Export permissions that user has

    Note: This is frontend-only permission checking. Backend API rules will be implemented in Task 2.

    3. Set up PocketBase API rules (backend):
       - Add to README/pb_schema setup instructions:
       - users collection rules (configure in PocketBase admin):
         - Create: Only Admin can create
         - Read: Admin can read all, users can read own
         - Update: Admin can update all, users can update own (except role)
         - Delete: Only Admin can delete

       Example PocketBase rule for users collection:
       ```
       // Create rule (Admin only)
       @request.auth.role = "admin"

       // Read rule (Admin all, users own)
       id = @request.auth.id || @request.auth.role = "admin"

       // Update rule (Admin all, users own - no role change)
       (id = @request.auth.id && role != @request.data.role) || @request.auth.role = "admin"

       // Delete rule (Admin only)
       @request.auth.role = "admin"
       ```

    Important: Password field should be hashed by PocketBase automatically. When creating user, use password field directly.
