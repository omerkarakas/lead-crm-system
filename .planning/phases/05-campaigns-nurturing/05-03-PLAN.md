       // Enrollment CRUD
       enrollLeadInCampaign(pb, lead_id, campaign_id, sequence_id?): Enroll lead in campaign
       unenrollLeadFromCampaign(pb, enrollment_id): Unsubscribe lead from campaign
       getLeadEnrollments(pb, lead_id): Get all enrollments for lead
       getCampaignEnrollments(pb, campaign_id): Get all enrollments for campaign

       // Eligibility checking
       checkEnrollmentEligibility(pb, lead_id, campaign_id): Check if lead can enroll
       isAlreadyEnrolled(pb, lead_id, campaign_id): Check for existing enrollment

       // Auto-enrollment

       autoEnrollLead(pb, lead): Enroll lead in all eligible campaigns
       evaluateAndEnroll(pb, lead_id): Re-evaluate and enroll after data changes

       // Status updates
       advanceEnrollmentStep(pb, enrollment_id): Move to next step
       completeEnrollment(pb, enrollment_id): Mark as completed
       failEnrollment(pb, enrollment_id, reason): Mark as failed

       // Unsubscribe
       generateUnsubscribeToken(): Generate random 32-char token
       unsubscribeByToken(pb, token, campaign_ids?): Unsubscribe via public link
