    1. Create types/campaign.ts with:

       // Enums
       enum CampaignType { Email = 'email', WhatsApp = 'whatsapp' }
       enum StepType { Email = 'email', WhatsApp = 'whatsapp', Delay = 'delay' }
       enum DelayType { Relative = 'relative', Absolute = 'absolute' }
       enum SegmentOperator { And = 'and', Or = 'or' }
       enum RuleOperator { Eq = 'eq', Ne = 'ne', Gt = 'gt', Lt = 'lt', Gte = 'gte', Lte = 'lte', Contains = 'contains', NotContains = 'not_contains' }

       // Audience Segment
       interface SegmentRule { field: string; operator: RuleOperator; value: any }
       interface AudienceSegment { operator: SegmentOperator; rules: SegmentRule[] }

       // Sequence Steps
       interface CampaignStep { id: string; order: number; type: StepType; template_id?: string; delay_minutes?: number; delay_type?: DelayType; scheduled_time?: string }

       // Campaign
       interface Campaign { id: string; name: string; description: string; type: CampaignType; audience_segment: AudienceSegment; auto_enroll_min_score?: number; is_active: boolean; created: string; updated: string }

       // Sequence
       interface Sequence { id: string; campaign_id: string; name: string; steps: CampaignStep[]; is_active: boolean; created: string; updated: string }

       // DTOs
       interface CreateCampaignDto { name: string; description: string; type: CampaignType; audience_segment: AudienceSegment; auto_enroll_min_score?: number; is_active?: boolean }
       interface UpdateCampaignDto { name?: string; description?: string; audience_segment?: AudienceSegment; auto_enroll_min_score?: number; is_active?: boolean }
       interface CreateSequenceDto { campaign_id: string; name: string; steps: CampaignStep[]; is_active?: boolean }
       interface UpdateSequenceDto { name?: string; steps?: CampaignStep[]; is_active?: boolean }

       // Segment Preview
       interface SegmentPreview { count: number; sample_leads: {id: string; name: string; status: string; score: number}[] }

  </action>
