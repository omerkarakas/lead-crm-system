    1. Create lib/api/campaign-analytics.ts with metrics functions:

       // Campaign-level metrics
       async function getCampaignMetrics(pb, campaign_id, date_range): Overall metrics
       Returns: { total_enrollments, active_enrollments, completed_enrollments, failed_enrollments, total_messages, sent_messages, delivered_messages, failed_messages }

       // Enrollment metrics
       async function getEnrollmentMetrics(pb, campaign_id, date_range): Enrollment funnel
       Returns: { enrollment_rate, completion_rate, failure_rate, avg_completion_time, step_completion_rates }

       // Email engagement
       async function getEmailEngagement(pb, campaign_id, date_range): Email metrics
       Returns: { total_sent, opens, clicks, open_rate, click_rate, click_to_open_rate }
       Note: For now, use estimated data (Resend doesn't provide open tracking in basic tier)

       // WhatsApp delivery
       async function getWhatsAppDelivery(pb, campaign_id, date_range): WhatsApp metrics
       Returns: { total_sent, delivered, read, failed, delivery_rate, read_rate }
       Source: sequence_messages status + WhatsApp message status

       // Conversion metrics
       async function getConversionMetrics(pb, campaign_id, date_range): Lead status changes
       Returns: { converted_to_customer, converted_to_booked, still_new, lost, conversion_rate }
       Query leads table for status changes during campaign period

       // Time series data
       async function getTimeSeriesData(pb, campaign_id, date_range, granularity): Data points over time
       Returns: { labels: string[], datasets: [{label, data}] }
       Granularity: day, week, month

       // Lead-level performance
       async function getLeadPerformance(pb, campaign_id, date_range): Per-lead metrics
       Returns: Array of { lead_id, lead_name, status, enrolled_at, completed_at, steps_completed, current_step }

       // Helper functions
       function filterByDateRange(query, start_date, end_date): Apply date filter
       function calculateRate(numerator, denominator): Safe division with 0 handling
       function formatDateForChart(date): Format date for chart labels

    2. Use PocketBase aggregations where possible for performance
    3. Cache results in memory for 5 minutes (same request)
